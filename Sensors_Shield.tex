	\caption[Caption]
	{
		\newline
		\textbf{Mount the AlphaSense sensor to the 3D printed mount  --- }
		\begin{itemize}
			\item Use 2 M3x10 screws to attach AlphaSense PM sensor on top of 3D printed mount
			\item There are 2 screw holes in the rectangular holders at the top of the mount (see picture) 
			\item Fit the PM sensor in between those rectangles aligning the brass screw holes on the sensor 
			\item The fan will be pointing out - opposite the direction of the stand base  
		\end{itemize}
	}
\end{SCfigure}
