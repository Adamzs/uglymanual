	\caption[Caption]
	{
		\newline
		\textbf{Mount the camera and lightsense board  --- }
		\begin{itemize}
			\item Connect lightsense board to outside of 3D printed mount and camera to inside facing out through hole in lightsense board, make sure camera is facing down toward base of the mount (the USB connector should be at the top by the RJ45 connector).
			\item Thread the screws through lightsense board, then the 3D printed mount, then camera (putting black washers at both ends) with nuts on camera side.
			\item The order is as follows: screw, washer, lightsense board, 3D mount, camera board, washer, hex nut.
		\end{itemize}
	}
